4%* cstutilnobs *;
5%* *;
6%* Returns the number of observations in a data set or an error. *;
7%* *;
8%* If the data set does not exist or cannot be opened, an error occurs. *;
9%* *;
10%* These actions occur: *;
11%* 1. The macro attempts to open the data set. *;
12%* If the data set cannot be opened, an error message is returned and *;
13%* processing stops. *;
14%* 2. The macro checks the values of the data set attributes ANOBS (Does SAS know *;
15%* how many observations there are?) and WHSTMT (Is a WHERE statement in *;
16%* effect?). *;
17%* 3. If SAS knows the number of observations and there is no WHERE clause, the *;
18%* value of the data set attribute NLOBS (number of logical observations) is *;
19%* returned. *;
20%* 4. If SAS does not know the number of observations (perhaps this is a view or *;
21%* transport data set) or if a WHERE clause is in effect, a forced read is done*;
22%* of the data set using NLOBSF. (NOTE: that this can be slow for large data *;
23%* sets.) *;
24%* If the data set exists, the value returned is a whole number. If the data *;
25%* set cannot be opened, a period (the default missing value) is returned. *;
26%* *;
27%* Examples: *;
28%* %put %cstutilnobs(_cstDataSetName=sashelp.class) *;
29%* %put %cstutilnobs(_cstDataSetName=sashelp.class(where=(sex="M"))) *;
30%* %put %cstutilnobs(_cstDataSetName=sashelp.vtable) *;
31%* title "sashelp.class has %cstutilnobs(_cstDataSetName=sashelp.class) obs" *;
32%* *;
33%* @param _cstDataSetName - required - The (libname.)memname of the data set. *;
34%* Default: _last_ *;
35%* *;
36%* @since 1.5 *;
37%* @exposure internal *;
38
39%macro cstutilnobs(_cstDataSetName=_last_)
40 / des='CST: Returns the number of observations in a data set';
41
42 %* Declare local variables used in the macro *;
43 %local _cstdsid _cstnobs _cstrc;
44
45 %let _cstdsid = %sysfunc(open(&_cstDataSetName, I));
46 %* Failed to open the data set *;
47 %if &_cstdsid = 0
48 %then %do;
49 %put %sysfunc(sysmsg());
50 %let _cstnobs = .;
51 %goto exit_abort;
52 %end;
53 %else %do;
54 %* Check to see if the engine is aware of # obs *;
55 %* and if there is an active where clause *;
56 %if %sysfunc(attrn(&_cstdsid, ANOBS)) = 1 and
57 %sysfunc(attrn(&_cstdsid, WHSTMT)) = 0
58 %then %let _cstnobs = %sysfunc(attrn(&_cstdsid, NLOBS));
59 %else %let _cstnobs = %sysfunc(attrn(&_cstdsid,NLOBSF));
60 %let _cstrc = %sysfunc(close(&_cstdsid));
61 %end;
62 %exit_abort:
63 %*;&_cstnobs%*;
64
65%mend cstutilnobs;
