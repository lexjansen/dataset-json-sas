4%* cstutilcheckvarsexist *;
5%* *;
6%* Determines whether a list of variables exists in a data set. *;
7%* *;
8%* This macro returns 1 when all variables exist and returns 0 when at least one *;
9%* variable does not exist or when a variable is not a valid SAS name. *;
10%* *;
11%* If the data set does not exist or cannot be opened, an error occurs. *;
12%* *;
13%* @param _cstDataSetName - required - The (libname.)memname of the data set. *;
14%* Default: _last_ *;
15%* @param _cstVarList - required - The list of blank-separated variables to check.*;
16%* @param _cstNotExistVarList - optional - An existing macro variable to *;
17%* contain the variables that could not be found in the data set. *;
18%* *;
19%* @history 2024-07-24 Made i macro variable local *;
20%* *;
21%* @since 1.7 *;
22%* @exposure external *;
23
24%macro cstutilcheckvarsexist(
25 _cstDataSetName=_last_,
26 _cstVarList=,
27 _cstNotExistVarList=
28 ) / des = 'CST: Check if variables exist in data set';
29
30 %local i _cstdsid _cstrc _cst_util_exists _cst_util_var _cst_util_noexistlist;
31
32 %****************************************************;
33 %* Check for missing parameters that are required *;
34 %****************************************************;
35 %if %sysevalf(%superq(_cstDataSetName)=,boolean) or
36 %sysevalf(%superq(_cstVarList)=,boolean) %then
37 %do;
38 %put %str(ER)ROR: [CSTLOG%str(MESSAGE).&sysmacroname] One or more REQUIRED parameters %str
39 ()(_cstDataSetName or _cstVarList) are missing.;
40 %let _cst_util_noexistlist=.;
41 %goto exit_abort;
42 %end;
43
44 %let _cstdsid = %sysfunc(open(&_cstDataSetName, I));
45 %* Failed to open the data set *;
46 %if &_cstdsid = 0
47 %then %do;
48 %put %sysfunc(sysmsg());
49 %let _cst_util_exists=;
50 %goto exit_abort;
51 %end;
52 %else %do;
53
54 %let _cst_util_noexistlist=;
55 %do i=1 %to %sysfunc(countw(&_cstVarList, %str( )));
56
57 %let _cst_util_var=%kscan(&_cstVarList, &i, %str( ));
58
59 %if %sysfunc(nvalid(&_cst_util_var))=0 %then %do;
60 %let _cst_util_noexistlist=&_cst_util_noexistlist &_cst_util_var;
61 %put [CSTLOG%str(MESSAGE).&sysmacroname] WAR%str(NING): &_cst_util_var is not a valid SAS variable name.;
62 %end;
63 %else
64 %if %sysfunc(varnum(&_cstdsid,&_cst_util_var)) lt 1 %then
65 %let _cst_util_noexistlist=&_cst_util_noexistlist &_cst_util_var;
66 %end;
67 %if %sysevalf(%superq(_cst_util_noexistlist)=, boolean)
68 %then %let _cst_util_exists=1;
69 %else %let _cst_util_exists=0;
70
71 %let _cstrc = %sysfunc(close(&_cstdsid));
72 %end;
73
74 %if %sysevalf(%superq(_cstNotExistVarList)=, boolean)=0 %then %do;
75 %if %symexist(&_cstNotExistVarList) %then %let &_cstNotExistVarList=&_cst_util_noexistlist;
76 %end;
77
78 %exit_abort:
79 %*;&_cst_util_exists%*;
80
81%mend cstutilcheckvarsexist;
