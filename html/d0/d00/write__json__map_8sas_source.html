1%macro write_json_map(fref=,dsds=,vards=);

3/***********************************************************************************\
4* *
5* Name: write_json_map.sas *
6* *
7* Description: Reads the mapped dataset and variable definitions from the specified *
8* input datasets and writes a JSON libname engine mapping to the *
9* specified file reference. *
10* *
11* Input parameters: *
12* *
13* FREF: The fileref for the JSON mapping. *
14* *
15* DSDS: The name of the dataset containing mapped dataset definitions. *
16* *
17* VARDS: The name of the dataset containing mapped dataset variable definitions. *
18* *
19\***********************************************************************************/
20
21 %local tdsid vdsid rc vrc;
22
23 proc json out=&fref pretty nosastags;
24 /* Initialize the mapping */
25 write open object;
26 write values "DATASETS";
27 /* Open the specified mapped dataset definition dataset */
28 write open array;
29 %let tdsid = %sysfunc(open(&dsds,i));
30 %syscall set(tdsid);
31 /* Loop through each of the mapped dataset definitions and ... */
32 %let rc=%sysfunc(fetch(&tdsid));
33 %do %while(&rc eq 0);
34 /* Add the dataset definition details to the mapping */
35 write open object;
36 write values "DSNAME" "&DSNAME";
37 write values "TABLEPATH" "&TABLEPATH";
38 write values "VARIABLES";
39 /* Open the mapped dataset variables dataset for the mapped dataset being processed and ... */
40 write open array;
41 %let vdsid = %sysfunc(open(&vards (where = (ordinal_datasets eq &ordinal_datasets)),i));
42 %syscall set(vdsid);
43 /* Loop through each of the mapped dataset variable definitions and ... */
44 %let vrc=%sysfunc(fetch(&vdsid));
45 %do %while(&vrc eq 0);
46 /* Add the variable definition details to the mapping */
47 write open object;
48 write values "NAME" "&NAME";
49 write values "TYPE" "&TYPE";
50 write values "PATH" "&PATH";
51 %if &CURRENT_LENGTH ne . %then
52 write values "CURRENT_LENGTH" &CURRENT_LENGTH;;
53 %if &INFORMAT ne %then %do;
54 write values "INFORMAT";
55 write open array;
56 %let _informat=%scan(&INFORMAT, 1, %str( ));
57 %let _length=%scan(&INFORMAT, 2, %str( ));
58 %let _width=%scan(&INFORMAT, 3, %str( ));
59 write values &_informat &_length &_width;
60 write close;
61 %end;
62 write close;
63 %let vrc=%sysfunc(fetch(&vdsid));
64 %end;
65 %let vrc = %sysfunc(close(&vdsid));
66 write close;
67 write close;
68 %let rc=%sysfunc(fetch(&tdsid));
69 %end;
70 %let rc = %sysfunc(close(&tdsid));
71 write close;
72 write close;
73 run;
74
75%mend;
