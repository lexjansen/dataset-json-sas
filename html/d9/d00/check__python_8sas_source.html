1/**
2 @file check_python.sas
3 @brief Checks that your SAS environment is able to run Python objects.
4
5 @details
6 This macro checks that your SAS environment is able to run Python objects.
7 Check the programs/config.sas file for the Python configuration.
8
9 It will check that the following environment variables have been defined
10 and reference existing files:
11 @li MAS_PYPATH
12 @li MAS_M2PATH
13
14 Python objects require environment variables to be set before you can use Python objects in your SAS environment.
15 If the environment variables have not been set, or if they have been set incorrectly,
16 SAS returns an error when you publish your Python code. Environment variable related errors can look like these examples:
17
18 ERROR: MAS_PYPATH environment variable is undefined.
19 ERROR: The executable /file-path/python.exe cannot be located or is not a valid executable.
20
21 Python code may require specific Python packages to be in your Python environment.
22 For example, the validate_datasetjson.sas macro assumes that your Python environment has the following packages:
23 - json
24 - jsonschema
25
26 More information:
27 Using PROC FCMP Python Objects:
28 https://documentation.sas.com/doc/en/pgmsascdc/9.4_3.5/lecompobjref/p18qp136f91aaqn1h54v3b6pkant.htm
29
30 Configuring SAS to Run the Python Language:
31 https://go.documentation.sas.com/doc/en/bicdc/9.4/biasag/n1mquxnfmfu83en1if8icqmx8cdf.htm
32
33 @author Lex Jansen
34
35**/
36%macro check_python() / des = 'Checks that your SAS environment is able to run Python objects';
37
38 %local
39 _MAS_PYPATH
40 _MAS_M2PATH
41 ;
42
43 %if %symexist(python_installed) %then %do;
44 %end;
45 %else %do;
46 %global python_installed;
47 %end;
48 %let python_installed = 0;
49
50 %let _MAS_PYPATH = %sysget(MAS_PYPATH);
51 %let _MAS_M2PATH = %sysget(MAS_M2PATH);
52
53 %if %sysevalf(%superq(_MAS_PYPATH)=, boolean) %then %do;
54 %put ERR%str(OR): [&sysmacroname] Environment variable MAS_PYPATH has not been defined. %str
55 ()As a result, Python scripts cannot be executed.;
56 %goto exit_macro;
57 %end;
58 %if %sysevalf(%superq(_MAS_M2PATH)=, boolean) %then %do;
59 %put ERR%str(OR): [&sysmacroname] Environment variable MAS_M2PATH has not been defined. %str
60 ()As a result, Python scripts cannot be executed.;
61 %goto exit_macro;
62 %end;
63
64 %if %sysfunc(fileexist(&_MAS_PYPATH)) = 0 %then %do;
65 %put ERR%str(OR): [&sysmacroname] File &_MAS_PYPATH does not exist. %str
66 ()As a result, Python scripts cannot be executed.;
67 %goto exit_macro;
68 %end;
69
70 %if %sysfunc(fileexist(&_MAS_M2PATH)) = 0 %then %do;
71 %put ERR%str(OR): [&sysmacroname] File &_MAS_M2PATH does not exist. %str
72 ()As a result, Python scripts cannot be executed.;
73 %goto exit_macro;
74 %end;
75
76 %put MAS_PYPATH = &_MAS_PYPATH;
77 %put MAS_M2PATH = &_MAS_M2PATH;
78
79 %let python_installed = 1;
80
81 %exit_macro:
82
83 %put &=python_installed;
84
85%mend check_python;
