4%* cstutilxptwrite *;
5%* *;
6%* Creates SAS Version 5 XPORT files from a library of SAS data sets. *;
7%* *;
8%* This macro creates SAS Version 5 XPORT files from a library of SAS data sets. *;
9%* *;
10%* Notes: *;
11%* 1. Any librefs referenced in macro parameters must be pre-allocated. *;
12%* 2. All SAS data sets are located in one library. *;
13%* 3. Existing SAS Version 5 XPORT files are overwritten. *;
14%* *;
15%* @macvar _cst_rc Task error status *;
16%* @macvar _cst_rcmsg Message associated with _cst_rc *;
17%* *;
18%* @param _cstSourceLibrary - required - The libref of the data folder/library *;
19%* in which the SAS data files are located. *;
20%* @param _cstOutputFolder - required - The folder in which to create the SAS *;
21%* Version 5 XPORT (XPT) files. *;
22%* @param _cstOptions - optional - Extra options for the PROC COPY statement. *;
23%* @param _cstReturn - required - The macro variable that contains the return *;
24%* value as set by this macro. *;
25%* Default: _cst_rc *;
26%* @param _cstReturnMsg - required - The macro variable that contains the return *;
27%* message as set by this macro. *;
28%* Default: _cst_rcmsg *;
29%* *;
30%* @since 1.7 *;
31%* @exposure external *;
32
33%macro cstutilxptwrite(
34 _cstSourceLibrary=,
35 _cstOutputFolder=,
36 _cstOptions=,
37 _cstReturn=_cst_rc,
38 _cstReturnMsg=_cst_rcmsg
39 ) / des='CST: Create XPT files from SAS Data Sets';
40
41 %local
42 _cstSrcMacro
43 _cstDatasets
44 _cstCounter
45 _cstRandom
46 dsid
47 ;
48
49 %let _cstSrcMacro=&SYSMACRONAME;
50 %let _cstRandom=%sysfunc(putn(%sysevalf(%sysfunc(ranuni(0))*10000,floor),z4.));
51
52 %***************************************************;
53 %* Check _cstReturn and _cstReturnMsg parameters *;
54 %***************************************************;
55 %if (%length(&_cstReturn)=0) or (%length(&_cstReturnMsg)=0) %then
56 %do;
57 %* We are not able to communicate other than to the LOG;
58 %put [CSTLOG%str(MESSAGE).&sysmacroname] ERR%str(OR): %str
59 ()macro parameters _CSTRETURN and _CSTRETURNMSG can not be missing.;
60 %goto exit_macro;
61 %end;
62
63 %if (%eval(not %symexist(&_cstReturn))) %then %global &_cstReturn;
64 %if (%eval(not %symexist(&_cstReturnMsg))) %then %global &_cstReturnMsg;
65
66 %*************************************************;
67 %* Set _cstReturn and _cstReturnMsg parameters *;
68 %*************************************************;
69 %let &_cstReturn=0;
70 %let &_cstReturnMsg=;
71
72 %************************;
73 %* Parameter checking *;
74 %************************;
75
76 %if %sysevalf(%superq(_cstSourceLibrary)=, boolean) %then
77 %do;
78 %* Rule: _cstSourceLibrary must be specified *;
79 %let &_cstReturn=1;
80 %let &_cstReturnMsg=_cstSourceLibrary must be specified.;
81 %goto exit_error;
82 %end;
83
84 %if %sysevalf(%superq(_cstOutputFolder)=, boolean) %then
85 %do;
86 %* Rule: _cstOutputFolder must be specified *;
87 %let &_cstReturn=1;
88 %let &_cstReturnMsg=_cstOutputFolder must be specified.;
89 %goto exit_error;
90 %end;
91
92 %if %sysfunc(libref(&_cstSourceLibrary)) %then
93 %do;
94 %* Rule: If _cstSourceLibrary is specified, it must exist *;
95 %let &_cstReturn=1;
96 %let &_cstReturnMsg=The libref _cstSourceLibrary=&_cstSourceLibrary has not been pre-allocated.;
97 %goto exit_error;
98 %end;
99
100 %if %sysfunc(filename(_cstDir,&_cstOutputFolder)) ne 0
101 %then %put %sysfunc(sysmsg());
102 %if not %sysfunc(fexist(&_cstDir)) %then
103 %do;
104 %* Rule: folder _cstOutputFolder must exist *;
105 %let &_cstReturn = 1;
106 %let &_cstReturnMsg = Specified folder _cstOutputFolder=&_cstOutputFolder does not exist;
107 %goto exit_error;
108 %end;
109
110 %************************;
111 %* Loop through library *;
112 %************************;
113
114 DATA _null_;
115 length mvname $8;
116 SET sashelp.vtable(where=(upcase(libname)="%upcase(&_cstSourceLibrary)")) end=end;
117 i+1;
118 mvname="ds"||left(put(i,2.));
119 call symputx(mvname,kcompress(klowcase(memname)));
120 if end then call symputx('_cstDatasets',kleft(put(_n_,2.)));
121 run;
122
123 %do _cstCounter=1 %to &_cstDatasets;
124
125 %if %sysfunc(libname(xpt&_cstRandom, &_cstOutputFolder/&&ds&_cstCounter...xpt, xport)) ne 0
126 %then %put %sysfunc(sysmsg());
127 %* libname xpt&_cstRandom xport "&_cstOutputFolder/&&ds&_cstCounter...xpt";
128 proc copy in=&_cstSourceLibrary OUT=xpt&_cstRandom &_cstOptions;
129 select &&ds&_cstCounter;
130 run;
131 %* libname xpt&_cstRandom clear;
132 %if %sysfunc(libname(xpt&_cstRandom)) ne 0
133 %then %put %sysfunc(sysmsg());
134
135 %end;
136
137 %****************************;
138 %* Handle any errors here *;
139 %****************************;
140%exit_error:
141
142 %if %length(&&&_cstReturnMsg)>0 %then
143 %put ERR%STR(OR): [CSTLOG%str(MESSAGE).&_cstSrcMacro] &&&_cstReturnMsg;
144
145%exit_macro:
146
147%mend cstutilxptwrite;
